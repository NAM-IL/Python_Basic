         {'name': 'Nate', 'midterm': 10, 'final': 30}]
df = pd.DataFrame(friend_dict_list, columns = ['name', 'midterm', 'final'])

df['average'] = df['total'] / 2

def hakjeum(average):
    grade ='F'
    if average >= 90:
        grade = 'A'
    elif average >= 80:
        grade = 'B'
    elif average >= 70:
        grade = 'C'
    else:
        grade = 'F'
    return grade

grade = [hakjeum(average) for average in list(df['average'])]
df['grade'] = grade
